### Component Architecture

**Home Page Structure:**

```typescript
interface HomePageProps {}

interface QuickLink {
  title: string;
  description: string;
  path: string;
  icon: React.ComponentType;
  count: number;
}

interface SiteStats {
  totalComponents: number;
  totalHooks: number;
  totalCategories: number;
}
```

**Statistics Fetching:**

```typescript
const useSiteStats = (): SiteStats => {
  const components = useComponents(); // From metadata
  const hooks = useHooks(); // From metadata

  return {
    totalComponents: components.length,
    totalHooks: hooks.length,
    totalCategories: new Set(components.map((c) => c.category)).size,
  };
};
```
